    def _make_plane(normal=(0, 0, 1), centroid=(0, 0, 5)):
        """Helper: build a horizontal Plane."""
        n = np.array(normal, dtype=float)
        n /= np.linalg.norm(n)
        if n[2] < 0:
            n = -n
        c = np.array(centroid, dtype=float)
        return Plane(normal=n, centroid=c, rms_residual=0.01, d=float(n @ c))

    ```python
    def build_feature_graph(
        polygons: dict[int, np.ndarray],
        planes: dict[int, Plane],
        tol: float = 1.0,
    ) -> dict:
        """Build feature graph from polygon vertices.

        Steps:
        1. Normalize winding to CCW
        2. Cluster vertices with three-pass tolerance
        3. Build nodes (one per cluster) with valence = number of distinct panels
        4. Build edges (one per panel pair sharing 2+ features)

        Returns dict matching INTG-02 schema:
        {
            "features": [{"id": int, "valence": int, "position_xyz": null, "panel_ids": [int]}],
            "edges": [{"panel_a": int, "panel_b": int, "feature_ids": [int]}]
        }
        """
    ```

    Implementation:
    1. Normalize winding: `normed = normalize_winding(polygons, planes)`
    2. Cluster: `groups, items = cluster_vertices(normed, planes, tol=tol)`
    3. Build features list: for each group (root -> members), create a feature dict:
       - `feature_id` = sequential integer starting at 0
       - `panel_ids` = sorted list of unique panel IDs from `items[m][0]` for `m` in members
       - `valence` = `len(panel_ids)`
       - `position_xyz` = `None` (solver is Phase 2)
       - Log classification: if valence == 2 then "corner", elif valence == 3 then "ridge_apex", elif valence >= 4 then "hip_apex"
    4. Build edges list: for each pair of panels (a, b) where a < b, collect all feature_ids where both panels appear in that feature's panel_ids. If the list has 2+ features, create an edge dict: `{"panel_a": a, "panel_b": b, "feature_ids": sorted(ids)}`
    5. Build result dict: `{"features": features, "edges": edges}`
    6. Log summary per D-03: valence distribution as `log.info("feature graph: %d nodes, %d edges", len(features), len(edges))` and per-valence counts.
    7. Return the dict.

    def print_dryrun(graph: dict) -> None:
        """Print feature graph JSON to stdout, summary to stderr, then exit(0).

        Per D-01: JSON to stdout, human-readable summary to stderr.
        Per D-03: Summary shows total node/edge counts and valence distribution.
        """
        json.dump(graph, sys.stdout, indent=2)
        sys.stdout.write("\n")
        sys.stdout.flush()

        # Summary to stderr (per D-03)
        features = graph["features"]
        edges = graph["edges"]
        valence_counts: dict[int, int] = {}
        for f in features:
            v = f["valence"]
            valence_counts[v] = valence_counts.get(v, 0) + 1

        print(f"--- snap-v2 dry-run summary ---", file=sys.stderr)
        print(f"nodes: {len(features)}, edges: {len(edges)}", file=sys.stderr)
        for v in sorted(valence_counts):
            label = {1: "unshared", 2: "corner", 3: "ridge_apex"}.get(v, f"hip_apex(v{v})")
            print(f"  valence-{v} ({label}): {valence_counts[v]} nodes", file=sys.stderr)
        print(f"-------------------------------", file=sys.stderr)
